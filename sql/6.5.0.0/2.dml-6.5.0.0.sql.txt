declare

v_count: number;

begin

select count(*) into v_count from work_order_reasons where work_order_reason_anme = 'RA04- Reasignado por error de carga';

if v_count > 1

	DBMS_OUTPUT.Put_line ('ERROR: Ya existe el registro RA04- Reasignado por error de carga en la 	tabla work_order_reason');
else
	INSERT
	INTO work_order_reasons VALUES
  	(
    	SEQ_WORKORDER_REASONS.nextval ,
    	'RA04- Reasignado por error de carga',
    	'6434',
    	(SELECT id
    	FROM work_order_reason_categories
    	WHERE wo_reason_category_code = 'RA'),
    	'S',
    	1,
    	(SELECT id FROM work_order_status WHERE wo_state_code = 'RA')
  	);


end;
